73  	public static void setDriver(Browser browser){

75  		getAutoDriver().setDriver(browser);
76  		ClassPool.reset();
77  		Set<Class<?>> cls = ClassPool.getClassPool();
78  		for(Class<?>clazz:cls){
79  			if(clazz.isAnnotationPresent(Register.class)){
80  				try {
81  					ProxyRunnerListener.register((RunnerListener) clazz.newInstance());
82  				} catch (InstantiationException e) {
83  					// TODO Auto-generated catch block
84  					e.printStackTrace();
85  				} catch (IllegalAccessException e) {
86  					// TODO Auto-generated catch block
87  					e.printStackTrace();
88  				}
89  			}
90  		}
91
